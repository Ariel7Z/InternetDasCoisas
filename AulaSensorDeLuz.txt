#define LDR A0
#define LED 2
float adc_ldr = 0.0;

void setup()
{
  	Serial.begin(9600);
  	pinMode(LED,OUTPUT);
  	digitalWrite(LED,LOW);
}

void loop()
{
  adc_ldr = analogRead(A0);
  Serial.println("Leitura LDR:");
  Serial.println(adc_ldr);
  
  if(adc_ldr >= 700)
  {
    digitalWrite(LED,HIGH);
  }
  else
  {
    digitalWrite(LED,LOW);
  }
}
